> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working DORS proof of concept — a reusable library plus a web-based key-compromise demo integrated into the existing mcp-i-room-booking app.

**Architecture:** The DORS library lives in `dors/src/` with zero framework dependencies (only `jose` for JWT). The room booking app at `mcp-i-room-booking/` imports it as `"dors": "file:../dors"` and adds a DORS service (Express on :3002), modifies the consent server, web server, MCP server, and web UI to demonstrate the full DORS flow including a "Simulate Key Compromise" button.

**Tech Stack:** TypeScript, Node.js, Express 5, `jose` (JWT), Ed25519, `@mcp-i/core`

import { createHash, randomUUID } from 'node:crypto';
import type { MCPIDORSStatusEntry, IssuanceRecord } from './types.js';

/**
 * Compute the credential hash per Section 8.1:
 * base64url( SHA-256( ASCII( compactJws ) ) )
 */
export function computeCredentialHash(compactJws: string): string {
  const digest = createHash('sha256').update(compactJws, 'ascii').digest();
  return digest.toString('base64url');
}

/**
 * Create a MCPIDORSStatusEntry for embedding in a VC.
 * Generates a random UUID v4 as the statusId.
 */
export function createDORSStatusEntry(statusService?: string): MCPIDORSStatusEntry {
  const entry: MCPIDORSStatusEntry = {
    type: 'MCPIDORSStatusEntry',
    statusId: `urn:uuid:${randomUUID()}`,
  };
  if (statusService) {
    entry.statusService = statusService;
  }
  return entry;
}

/**
 * Build an IssuanceRecord to register with the DORS responder.
 */
export function createIssuanceRecord(
  issuerDid: string,
  statusId: string,
  compactJws: string,
): IssuanceRecord {
  return {
    issuerDid,
    statusId,
    credentialHash: computeCredentialHash(compactJws),
    credentialHashAlg: 'sha-256',
    status: 'active',
    issuedAt: Math.floor(Date.now() / 1000),
  };
}

```typescript
import { randomUUID } from 'node:crypto';
import { SignJWT, importPKCS8, importJWK } from 'jose';
import type {
  DORSResponderConfig,
  DORSStatusQuery,
  DORSAssertionClaims,
  IssuanceRecord,
  DORSReasonCode,
} from './types.js';

export class DORSResponder {
  private records = new Map<string, IssuanceRecord>();
  private config: DORSResponderConfig;

  constructor(config: DORSResponderConfig) {
    this.config = config;
  }

  /** Register a new credential issuance. */
  registerIssuance(record: IssuanceRecord): void {
    this.records.set(record.statusId, { ...record });
  }

  /** Answer a status query — returns a signed compact JWS. */
  async queryStatus(query: DORSStatusQuery): Promise<string> {
    const record = this.records.get(query.status_id);

    let status: DORSAssertionClaims['status'];
    let reason: string | undefined;
    let statusChangedAt: number | undefined;

    if (!record || record.credentialHash !== query.credential_hash) {
      status = 'unknown';
    } else {
      status = record.status === 'active' ? 'active'
        : record.status === 'revoked' ? 'revoked'
        : record.status === 'suspended' ? 'suspended'
        : 'unknown';
      reason = record.reason;
      statusChangedAt = record.statusChangedAt;
    }

    const now = Math.floor(Date.now() / 1000);
    const claims: Record<string, unknown> = {
      sub: query.status_id,
      jti: `urn:uuid:${randomUUID()}`,
      credential_hash: query.credential_hash,
      credential_hash_alg: query.credential_hash_alg,
      status,
    };

    if (reason) claims['reason'] = reason;
    if (statusChangedAt) claims['status_changed_at'] = statusChangedAt;
    if (query.nonce) claims['nonce'] = query.nonce;

    const jwt = await new SignJWT(claims)
      .setProtectedHeader({
        alg: 'EdDSA',
        typ: 'mcpi-status+jwt',
        kid: this.config.statusAuthorityKid,
      })
      .setIssuer(this.config.statusAuthorityDid)
      .setIssuedAt(now)
      .setExpirationTime(now + this.config.defaultTtlSeconds)
      .sign(this.config.privateKey as CryptoKey);

    return jwt;
  }

  /** Revoke a single credential. */
  revoke(statusId: string, reason?: DORSReasonCode | string): void {
    const record = this.records.get(statusId);
    if (record) {
      record.status = 'revoked';
      record.reason = reason ?? 'unspecified';
      record.statusChangedAt = Math.floor(Date.now() / 1000);
    }
  }

  /** Bulk revoke all credentials for an issuer DID. Returns count of revoked credentials. */
  revokeByIssuer(issuerDid: string, reason: DORSReasonCode | string): number {
    let count = 0;
    const now = Math.floor(Date.now() / 1000);
    for (const record of this.records.values()) {
      if (record.issuerDid === issuerDid && record.status !== 'revoked') {
        record.status = 'revoked';
        record.reason = reason;
        record.statusChangedAt = now;
        count++;
      }
    }
    return count;
  }

  /** Suspend a credential. */
  suspend(statusId: string): void {
    const record = this.records.get(statusId);
    if (record && record.status === 'active') {
      record.status = 'suspended';
      record.reason = 'administrative_suspend';
      record.statusChangedAt = Math.floor(Date.now() / 1000);
    }
  }

  /** Unsuspend a credential (return to active). */
  unsuspend(statusId: string): void {
    const record = this.records.get(statusId);
    if (record && record.status === 'suspended') {
      record.status = 'active';
      record.reason = undefined;
      record.statusChangedAt = Math.floor(Date.now() / 1000);
    }
  }

  /** Get all issuance records (for debug/admin). */
  getRecords(): IssuanceRecord[] {
    return Array.from(this.records.values());
  }
}
```

```typescript
import { jwtVerify } from 'jose';
import { computeCredentialHash } from './issuer.js';
import type {
  DORSAssertionClaims,
  DORSVerifierPolicy,
  DORSVerificationResult,
  DORSStatusQuery,
} from './types.js';

/**
 * Verify a DORS status assertion JWT per Section 12.4.
 */
export async function verifyDORSAssertion(config: {
  assertionJwt: string;
  expectedStatusId: string;
  credentialCompactJws: string;
  statusAuthorityPublicKey: CryptoKey;
  statusAuthorityDid: string;
  policy: DORSVerifierPolicy;
}): Promise<DORSVerificationResult> {
  try {
    // Step 1: Verify JWS signature and decode
    const { payload, protectedHeader } = await jwtVerify(
      config.assertionJwt,
      config.statusAuthorityPublicKey,
    );

    // Step 1: Check typ
    if (protectedHeader.typ !== 'mcpi-status+jwt') {
      return { valid: false, error: `Invalid typ: expected "mcpi-status+jwt", got "${protectedHeader.typ}"` };
    }

    const claims = payload as unknown as DORSAssertionClaims;

    // Step 3: Check iss
    if (claims.iss !== config.statusAuthorityDid) {
      return { valid: false, error: `iss mismatch: expected "${config.statusAuthorityDid}", got "${claims.iss}"` };
    }

    // Step 4: Check sub
    if (claims.sub !== config.expectedStatusId) {
      return { valid: false, error: `sub mismatch: expected "${config.expectedStatusId}", got "${claims.sub}"` };
    }

    // Step 5: Check credential_hash_alg
    if (claims.credential_hash_alg !== 'sha-256') {
      return { valid: false, error: `Unsupported credential_hash_alg: "${claims.credential_hash_alg}"` };
    }

    // Step 6: Compute and check credential_hash
    const expectedHash = computeCredentialHash(config.credentialCompactJws);
    if (claims.credential_hash !== expectedHash) {
      return { valid: false, error: 'credential_hash mismatch' };
    }

    // Step 7: Check exp (jose already checks this, but be explicit)
    const now = Math.floor(Date.now() / 1000);
    if (claims.exp <= now) {
      return { valid: false, error: 'DORS assertion expired' };
    }

    // Step 8: Check iat not in the future (with clock skew)
    if (claims.iat > now + config.policy.allowedClockSkewSeconds) {
      return { valid: false, error: 'DORS assertion iat is in the future beyond allowed clock skew' };
    }

    // Step 9: Check TTL bounds
    const ttl = claims.exp - claims.iat;
    if (ttl > config.policy.maxTtlSeconds) {
      return { valid: false, error: `DORS assertion TTL (${ttl}s) exceeds policy max (${config.policy.maxTtlSeconds}s)` };
    }

    return { valid: true, claims };
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    return { valid: false, error: `DORS assertion verification failed: ${message}` };
  }
}

/**
 * Fetch a live DORS assertion from a service endpoint (Section 12.5).
 */
export async function fetchDORSAssertion(config: {
  serviceEndpoint: string;
  statusId: string;
  credentialCompactJws: string;
  nonce?: string;
}): Promise<string> {
  const body: DORSStatusQuery = {
    status_id: config.statusId,
    credential_hash: computeCredentialHash(config.credentialCompactJws),
    credential_hash_alg: 'sha-256',
  };
  if (config.nonce) {
    body.nonce = config.nonce;
  }

  const response = await fetch(config.serviceEndpoint, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/jwt',
    },
    body: JSON.stringify(body),
  });

  if (!response.ok) {
    throw new Error(`DORS service returned ${response.status}: ${await response.text()}`);
  }

  return await response.text();
}

import express from 'express';
import { generateKeyPair, exportJWK } from 'jose';
import {
  DORSResponder,
  type IssuanceRecord,
  type DORSStatusQuery,
} from 'dors';

const PORT = parseInt(process.env['DORS_PORT'] ?? '3002', 10);

async function main() {
  // Generate Ed25519 key pair for the status authority (independent from consent server)
  const { publicKey, privateKey } = await generateKeyPair('EdDSA', { crv: 'Ed25519' });
  const publicKeyJwk = await exportJWK(publicKey);

  // Derive a simple DID for the status authority
  const statusAuthorityDid = `did:key:dors-status-authority-${Date.now()}`;
  const statusAuthorityKid = `${statusAuthorityDid}#key-1`;

  console.log(`[dors] Status Authority DID: ${statusAuthorityDid}`);

  const responder = new DORSResponder({
    statusAuthorityDid,
    statusAuthorityKid,
    privateKey,
    defaultTtlSeconds: 300, // 5 minutes
  });

  const app = express();
  app.use(express.json());

  // CORS for web UI
  app.use((_req, res, next) => {
    res.setHeader('Access-Control-Allow-Origin', '*');
    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Accept');
    if (_req.method === 'OPTIONS') { res.sendStatus(204); return; }
    next();
  });

  // Spec Section 9: Status query
  app.post('/v1/status', async (req, res) => {
    const query = req.body as DORSStatusQuery;

    if (!query.status_id || !query.credential_hash || !query.credential_hash_alg) {
      res.status(400).json({ error: 'Missing required fields: status_id, credential_hash, credential_hash_alg' });
      return;
    }

    if (query.credential_hash_alg !== 'sha-256') {
      res.status(400).json({ error: 'Unsupported credential_hash_alg. Must be "sha-256".' });
      return;
    }

    const jwt = await responder.queryStatus(query);
    res.setHeader('Content-Type', 'application/jwt');
    res.send(jwt);
  });

  // Admin: register issuance
  app.post('/admin/register', (req, res) => {
    const record = req.body as IssuanceRecord;
    responder.registerIssuance(record);
    console.log(`[dors] Registered issuance: statusId=${record.statusId}`);
    res.json({ ok: true });
  });

  // Admin: revoke single credential
  app.post('/admin/revoke', (req, res) => {
    const { status_id, reason } = req.body as { status_id: string; reason?: string };
    responder.revoke(status_id, reason);
    console.log(`[dors] Revoked: statusId=${status_id}, reason=${reason ?? 'unspecified'}`);
    res.json({ ok: true, status_id });
  });

  // Admin: bulk revoke by issuer DID (key compromise scenario)
  app.post('/admin/revoke-by-issuer', (req, res) => {
    const { issuer_did, reason } = req.body as { issuer_did: string; reason: string };
    const count = responder.revokeByIssuer(issuer_did, reason);
    console.log(`[dors] Bulk revoked ${count} credential(s) for issuer=${issuer_did}, reason=${reason}`);
    res.json({ ok: true, count });
  });

  // Admin: list all records (debug)
  app.get('/admin/records', (_req, res) => {
    res.json(responder.getRecords());
  });

  // Well-known: status authority info
  app.get('/.well-known/dors', (_req, res) => {
    res.json({
      statusAuthorityDid,
      statusAuthorityKid,
      publicKeyJwk,
    });
  });

  app.listen(PORT, () => {
    console.log(`[dors] DORS service: http://localhost:${PORT}`);
    console.log(`[dors] Status query: POST http://localhost:${PORT}/v1/status`);
    console.log(`[dors] Well-known:   GET  http://localhost:${PORT}/.well-known/dors`);
  });
}

main().catch(err => { console.error('Fatal:', err); process.exit(1); });
```
